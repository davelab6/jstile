== 必要な DAV の設定

<Directory "/var/www/html/jstile/data">
    Header add MS-Author-Via "DAV"
    DAV On
    AuthName "a WebDAV folder for jstile"
    AuthType Basic
    AuthUserFile /etc/httpd/conf/.dav
    <LimitExcept GET OPTIONS REPORT>
      Require valid-user
    </LimitExcept>
    Options Indexes
</Directory>
